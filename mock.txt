function timeConversion(s) {

    let hour = parseInt(s.slice(0, 2));
    let rest = s.slice(2, 8);
    let meridian = s.slice(8);

    if (meridian === "AM") {
        hour = (hour === 12) ? 0 : hour;
    } else {
        hour = (hour === 12) ? 12 : hour + 12;
    }

    return `${hour.toString().padStart(2, '0')}${rest}`;
}
